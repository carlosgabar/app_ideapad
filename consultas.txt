SELECT
    c.nombre,
    c.fecha_inicio,
    c.fecha_fin,
    STRING_AGG(t.descripcion, ', ') AS tareas
FROM
    curso c
INNER JOIN tarea t ON c.id_curso = t.id_curso
WHERE
    c.status = 'activo'
GROUP BY
    c.nombre, c.fecha_inicio, c.fecha_fin;
